  <script>

    const eventos = {

      vaga: {

        mensagem: "🎯 VAGA DETECTADA no site de agendamentos!",

        repeticoes: 10

      },

      logout: {

        mensagem: "🔐 Sessão expirada por inatividade.",

        repeticoes: 1

      },

      chutado: {

        mensagem: "🚫 Você foi CHUTADO do sistema por erro interno.",

        repeticoes: 1

      }

    };


    function obterParametro(nome) {

      const urlParams = new URLSearchParams(window.location.search);

      return urlParams.get(nome);

    }


    function enviarNotificacao(mensagem, repeticoes) {

      const url = "https://api.telegram.org/bot7284975115:AAHu8krgrGJXXW_GyTxUFbyDfnIVsQcdUdg/sendMessage";

      for (let i = 0; i < repeticoes; i++) {

        fetch(url, {

          method: "POST",

          headers: {

            "Content-Type": "application/json"

          },

          body: JSON.stringify({

            chat_id: "7631883392",

            text: mensagem

          })

        })

        .then(() => console.log(`📩 Mensagem ${i+1} enviada`))

        .catch(err => console.warn("❌ Erro ao enviar:", err));

      }

    }


    const tipoEvento = obterParametro('tipo');


    if (eventos[tipoEvento]) {

      const evento = eventos[tipoEvento];

      document.getElementById('mensagem').innerText = evento.mensagem;

      enviarNotificacao(evento.mensagem, evento.repeticoes);

    } else {

      document.getElementById('mensagem').innerText = "❓ Tipo de evento desconhecido.";

    }


    let timeoutId;

    let alertSent = false;


    function monitorScript() {

      const signal = localStorage.getItem('scriptRunning');


      if (signal === 'true') {

          console.log("🟢 Script está funcionando.");

          localStorage.setItem('scriptRunning', 'false');

          clearTimeout(timeoutId);

          timeoutId = setTimeout(monitorScript, 30000);

      } else {

          if (!alertSent) {

              alertSent = true;

              alert("⚠️ O script parou de rodar ou não está mais enviando o sinal.");

              console.log("🔴 O script não enviou o sinal. Enviando alerta.");

              window.location.href = "https://joaogqpugliese.github.io/alerta-eventos/alerta.html?tipo=chutado";

          }

      }

    }


    // Aguardar 20 segundos antes de iniciar o monitoramento

    window.onload = () => {

      setTimeout(monitorScript, 20000);

    };

  </script>
